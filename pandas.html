Contekan Manipulasi Data Pandas

Operasi paling umum untuk membersihkan, mengubah, dan menganalisis data.

Operasi Dasar & Seleksi Data

Membaca CSV & Melihat Data Awal

import pandas as pd

# Ganti 'nama_file.csv' dengan path file Anda
# df = pd.read_csv('nama_file.csv') 

# Contoh membuat DataFrame manual
data = {'Nama': ['Ana', 'Budi', 'Caca'], 'Usia': [28, 34, 29]}
df = pd.DataFrame(data)

# Melihat 5 baris pertama
print(df.head())

Seleksi Kolom

# data dari kartu sebelumnya
# df = ...

# Memilih satu kolom (hasilnya Series)
nama_series = df['Nama']
print(nama_series)

# Memilih beberapa kolom (hasilnya DataFrame)
subset_df = df[['Nama', 'Usia']]
print(subset_df)

Seleksi Baris (.loc & .iloc)

# .iloc memilih berdasarkan posisi integer (mulai dari 0)
baris_pertama = df.iloc[0] 
print(baris_pertama)

# .loc memilih berdasarkan label index
# Jika indexnya default, maka sama dengan .iloc
baris_kedua = df.loc[1]
print(baris_kedua)

Filtering Data

# data dari kartu sebelumnya
# df = ...

# Memilih baris dimana Usia di atas 30
diatas_30 = df[df['Usia'] > 30]
print(diatas_30)

# Kondisi ganda (usia > 25 DAN Nama adalah 'Caca')
kondisi_ganda = df[(df['Usia'] > 25) & (df['Nama'] == 'Caca')]
print(kondisi_ganda)

Group By & Agregasi

data = {'Tim': ['A','B','A','B'], 'Poin': [10,12,5,8]}
df_poin = pd.DataFrame(data)

# Kelompokkan berdasarkan 'Tim' lalu hitung total poin
total_poin_per_tim = df_poin.groupby('Tim')['Poin'].sum()
print(total_poin_per_tim)

Menambah & Menghapus Kolom

# data dari kartu pertama
# df = ...

# Menambah kolom baru
df['Pekerjaan'] = ['Dokter', 'Insinyur', 'Guru']
print("Setelah ditambah:")
print(df)

# Menghapus kolom 'Pekerjaan'
df = df.drop('Pekerjaan', axis=1)
print("\nSetelah dihapus:")
print(df)
