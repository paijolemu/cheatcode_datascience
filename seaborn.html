Plot Statistik Populer

Violin Plot

import seaborn as sns
import matplotlib.pyplot as plt

# Seaborn memiliki dataset bawaan
tips = sns.load_dataset("tips")

sns.violinplot(x="day", y="total_bill", data=tips, palette="muted")
plt.title('Distribusi Total Tagihan per Hari')
plt.show()

Heatmap Korelasi

import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# Membuat data acak dan matriks korelasi
data = pd.DataFrame(np.random.rand(10, 10))
correlation = data.corr()

sns.heatmap(correlation, annot=True, cmap='coolwarm', fmt=".1f")
plt.title('Heatmap Matriks Korelasi')
plt.show()

Distribution Plot

import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")

# Menampilkan histogram dengan kurva kepadatan (KDE)
sns.histplot(data=tips, x="total_bill", kde=True, color="teal")
plt.title('Distribusi Total Tagihan')
plt.show()

Pair Plot

import seaborn as sns
import matplotlib.pyplot as plt

# Menggunakan dataset 'iris' yang terkenal
iris = sns.load_dataset("iris")

# Membuat scatter plot untuk setiap pasangan variabel
# dan memberi warna berdasarkan spesies
sns.pairplot(iris, hue="species", palette="viridis")
plt.suptitle('Analisis Hubungan Antar Variabel Iris', y=1.02)
plt.show()
