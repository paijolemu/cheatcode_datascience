Galeri Visualisasi Seaborn

Membuat plot statistik yang indah dengan kode yang lebih ringkas.

Plot Statistik Populer

Distribution Plot (histplot)

import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")
sns.histplot(data=tips, x="total_bill", kde=True, color="teal")
plt.title('Distribusi Total Tagihan')
plt.show()

Violin Plot

import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")
sns.violinplot(x="day", y="total_bill", data=tips, palette="muted")
plt.title('Distribusi Total Tagihan per Hari')
plt.show()

Box Plot

import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")
sns.boxplot(x="day", y="total_bill", hue="smoker", data=tips, palette="coolwarm")
plt.title('Box Plot Tagihan per Hari (Dipisah Perokok)')
plt.show()

Heatmap Korelasi

import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

# Membuat data acak
data = pd.DataFrame(np.random.rand(10, 10), columns=[f'Var{i}' for i in range(10)])
correlation = data.corr()

sns.heatmap(correlation, annot=True, cmap='viridis', fmt=".1f")
plt.title('Heatmap Matriks Korelasi')
plt.show()

Joint Plot

import seaborn as sns
import matplotlib.pyplot as plt

tips = sns.load_dataset("tips")
# Menampilkan scatter plot dengan distribusi di setiap sumbu
sns.jointplot(data=tips, x="total_bill", y="tip", kind="reg", color="purple")
plt.suptitle('Hubungan Tagihan dan Tip', y=1.02)
plt.show()

Pair Plot

import seaborn as sns
import matplotlib.pyplot as plt

iris = sns.load_dataset("iris")
# Membuat scatter plot untuk setiap pasangan variabel
sns.pairplot(iris, hue="species", palette="plasma")
plt.suptitle('Analisis Hubungan Antar Variabel Iris', y=1.02)
plt.show()
