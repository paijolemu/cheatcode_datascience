Matplotlib Cheats

Koleksi kode untuk membuat plot umum menggunakan Matplotlib.

Plot Populer

Line Plot

import matplotlib.pyplot as plt

tahun = [2018, 2019, 2020, 2021, 2022]
penjualan = [100, 120, 115, 150, 180]

plt.plot(tahun, penjualan, marker='o', linestyle='--')
plt.title('Tren Penjualan Tahunan')
plt.xlabel('Tahun')
plt.ylabel('Penjualan (dalam juta)')
plt.grid(True)
plt.show()

Histogram

import matplotlib.pyplot as plt
import numpy as np

# Data acak untuk contoh
data = np.random.normal(loc=70, scale=10, size=500)

plt.hist(data, bins=20, color='skyblue', edgecolor='black')
plt.title('Distribusi Nilai Ujian')
plt.xlabel('Nilai')
plt.ylabel('Frekuensi')
plt.show()

Pie Chart

import matplotlib.pyplot as plt

labels = ['Android', 'iOS', 'Lainnya']
market_share = [65, 30, 5]
explode = (0.05, 0, 0) # menonjolkan slice 'Android'

plt.pie(market_share, explode=explode, labels=labels, 
        autopct='%1.1f%%', shadow=True, startangle=140)
plt.axis('equal') # Agar pie chart terlihat bulat
plt.title('Pangsa Pasar Sistem Operasi Mobile')
plt.show()
