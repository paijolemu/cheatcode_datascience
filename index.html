Scatter Plot dengan Matplotlib


import matplotlib.pyplot as plt

x = [5, 7, 8, 7, 2, 17, 2, 9, 4]
y = [99, 86, 87, 88, 111, 86, 103, 87, 94]

plt.scatter(x, y)
plt.xlabel('Usia Mobil')
plt.ylabel('Kecepatan')
plt.title('Scatter Plot: Usia vs Kecepatan Mobil')
plt.show()
