Klasifikasi: Random Forest

# 1. Import library yang dibutuhkan
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import confusion_matrix, classification_report

# 2. Muat dan siapkan data
iris = load_iris()
X, y = iris.data, iris.target

# 3. Bagi data menjadi training dan testing
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42)

# 4. Inisialisasi dan latih model
model = RandomForestClassifier(n_estimators=100, random_state=42)
model.fit(X_train, y_train)

# 5. Lakukan prediksi pada data test
predictions = model.predict(X_test)

# 6. Evaluasi dan visualisasikan hasil
print(classification_report(y_test, predictions, target_names=iris.target_names))

cm = confusion_matrix(y_test, predictions)
sns.heatmap(cm, annot=True, cmap='Blues',
            xticklabels=iris.target_names,
            yticklabels=iris.target_names)
plt.title('Confusion Matrix - Random Forest')
plt.xlabel('Predicted')
plt.ylabel('Actual')
plt.show()

Regresi: Linear Regression

# 1. Import library
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_california_housing
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error

# 2. Muat dan siapkan data
housing = fetch_california_housing()
X, y = housing.data, housing.target

# 3. Bagi data
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42)

# 4. Inisialisasi dan latih model
model = LinearRegression()
model.fit(X_train, y_train)

# 5. Lakukan prediksi
predictions = model.predict(X_test)

# 6. Evaluasi dan visualisasikan
mse = mean_squared_error(y_test, predictions)
print(f"Mean Squared Error: {mse:.2f}")

plt.scatter(y_test, predictions, alpha=0.5)
plt.plot([y.min(), y.max()], [y.min(), y.max()], '--r', linewidth=2)
plt.title('Actual vs. Predicted Housing Prices')
plt.xlabel('Actual Prices')
plt.ylabel('Predicted Prices')
plt.show()

Clustering: K-Means

# 1. Import library
import matplotlib.pyplot as plt
from sklearn.datasets import make_blobs
from sklearn.cluster import KMeans

# 2. Buat data sintetis untuk clustering
X, _ = make_blobs(n_samples=300, centers=4, 
                  cluster_std=0.60, random_state=0)

# 3. Inisialisasi dan latih model K-Means
kmeans = KMeans(n_clusters=4, random_state=0, n_init=10)
kmeans.fit(X)

# 4. Dapatkan hasil cluster dan pusatnya
y_kmeans = kmeans.predict(X)
centers = kmeans.cluster_centers_

# 5. Visualisasikan hasil
plt.scatter(X[:, 0], X[:, 1], c=y_kmeans, s=50, cmap='viridis')
plt.scatter(centers[:, 0], centers[:, 1], c='red', s=200, alpha=0.75, marker='X')
plt.title('K-Means Clustering')
plt.show()

Reduksi Dimensi: PCA

# 1. Import library
import matplotlib.pyplot as plt
from sklearn.datasets import load_digits
from sklearn.decomposition import PCA

# 2. Muat data (64 dimensi)
digits = load_digits()
X = digits.data

# 3. Inisialisasi PCA untuk mereduksi menjadi 2 dimensi
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X)

# 4. Visualisasikan data setelah direduksi
plt.figure(figsize=(8, 6))
plt.scatter(X_pca[:, 0], X_pca[:, 1], c=digits.target,
            edgecolor='none', alpha=0.7, cmap='plasma')
plt.title('Visualisasi Data Digits dengan PCA')
plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.colorbar(label='Digit Label')
plt.show()
